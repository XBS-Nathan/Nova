# Interactive Init Selectors Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace free-text prompts in `nova init` with interactive selectors, using version lists fetched from Docker Hub at build time via `go generate`.

**Architecture:** A `go generate` program queries Docker Hub for PHP/Node/MySQL/Postgres/Redis tags, filters them to stable major.minor versions, and writes a generated Go file with version slices. `cmd/init.go` then uses `pterm.DefaultInteractiveSelect` for all option fields and `pterm.DefaultInteractiveTextInput` for free-text fields.

**Tech Stack:** Go, `pterm` (already a dependency), Docker Hub REST API (`hub.docker.com/v2/repositories/library/`)

```go
// internal/config/cmd/fetchversions/main.go
package main

import (
	"encoding/json"
	"fmt"
	"net/http"
	"os"
	"regexp"
	"sort"
	"strings"
	"time"
)

type tagResponse struct {
	Results []struct {
		Name string `json:"name"`
	} `json:"results"`
	Next string `json:"next"`
}

// fetchTags fetches up to maxPages of tags for a Docker Hub library image.
func fetchTags(image string, maxPages int) ([]string, error) {
	client := &http.Client{Timeout: 15 * time.Second}
	var tags []string
	url := fmt.Sprintf("https://hub.docker.com/v2/repositories/library/%s/tags?page_size=100", image)

	for page := 0; page < maxPages && url != ""; page++ {
		resp, err := client.Get(url)
		if err != nil {
			return nil, err
		}
		var data tagResponse
		if err := json.NewDecoder(resp.Body).Decode(&data); err != nil {
			resp.Body.Close()
			return nil, err
		}
		resp.Body.Close()
		for _, r := range data.Results {
			tags = append(tags, r.Name)
		}
		url = data.Next
	}
	return tags, nil
}

// filterPHP extracts major.minor from tags matching "X.Y-fpm" (exact, no suffix).
func filterPHP(tags []string) []string {
	re := regexp.MustCompile(`^(\d+\.\d+)-fpm$`)
	seen := map[string]bool{}
	var versions []string
	for _, t := range tags {
		if m := re.FindStringSubmatch(t); m != nil && !seen[m[1]] {
			seen[m[1]] = true
			versions = append(versions, m[1])
		}
	}
	sort.Sort(sort.Reverse(sort.StringSlice(versions)))
	return versions
}

// filterNode extracts major version from tags matching "X-alpine" (exact).
func filterNode(tags []string) []string {
	re := regexp.MustCompile(`^(\d+)-alpine$`)
	seen := map[string]bool{}
	var versions []string
	for _, t := range tags {
		if m := re.FindStringSubmatch(t); m != nil && !seen[m[1]] {
			seen[m[1]] = true
			versions = append(versions, m[1])
		}
	}
	sort.Sort(sort.Reverse(sort.StringSlice(versions)))
	return versions
}

// filterMajorMinor extracts tags matching bare "X.Y" (digits.digits, no suffix).
func filterMajorMinor(tags []string) []string {
	re := regexp.MustCompile(`^(\d+\.\d+)$`)
	seen := map[string]bool{}
	var versions []string
	for _, t := range tags {
		if m := re.FindStringSubmatch(t); m != nil && !seen[m[1]] {
			seen[m[1]] = true
			versions = append(versions, m[1])
		}
	}
	sort.Sort(sort.Reverse(sort.StringSlice(versions)))
	return versions
}

// filterMajor extracts tags matching bare "X" (single integer, no suffix).
func filterMajor(tags []string) []string {
	re := regexp.MustCompile(`^(\d+)$`)
	seen := map[string]bool{}
	var versions []string
	for _, t := range tags {
		if m := re.FindStringSubmatch(t); m != nil && !seen[m[1]] {
			seen[m[1]] = true
			versions = append(versions, m[1])
		}
	}
	sort.Sort(sort.Reverse(sort.StringSlice(versions)))
	return versions
}

type imageSpec struct {
	Name     string
	VarName  string
	Filter   func([]string) []string
	Fallback []string
}

func main() {
	specs := []imageSpec{
		{"php", "PHPVersions", filterPHP, []string{"8.4", "8.3", "8.2", "8.1"}},
		{"node", "NodeVersions", filterNode, []string{"22", "20", "18"}},
		{"mysql", "MySQLVersions", filterMajorMinor, []string{"9.0", "8.4", "8.0"}},
		{"postgres", "PostgresVersions", filterMajor, []string{"17", "16", "15"}},
		{"redis", "RedisVersions", filterMajor, []string{"8", "7"}},
	}

	var b strings.Builder
	b.WriteString("// Code generated by fetchversions; DO NOT EDIT.\n\n")
	b.WriteString("package config\n\n")

	for _, s := range specs {
		tags, err := fetchTags(s.Name, 3)
		versions := s.Fallback
		if err == nil {
			if filtered := s.Filter(tags); len(filtered) > 0 {
				versions = filtered
			}
		}
		fmt.Fprintf(&b, "var %s = []string{", s.VarName)
		for i, v := range versions {
			if i > 0 {
				b.WriteString(", ")
			}
			fmt.Fprintf(&b, "%q", v)
		}
		b.WriteString("}\n")
	}

	if err := os.WriteFile("versions_generated.go", b.Bytes(), 0644); err != nil {
		fmt.Fprintf(os.Stderr, "writing versions_generated.go: %v\n", err)
		os.Exit(1)
	}
	fmt.Println("generated versions_generated.go")
}
```

### Task 3: Rewrite `cmd/init.go` to use interactive selectors

**Files:**
- Modify: `cmd/init.go`

- [ ] **Step 1: Rewrite init.go with pterm selectors**

Replace the entire `initCmd.RunE` function body. Key changes:
- Remove `bufio` import and `prompt()` helper
- Add `pterm` import
- Use `pterm.DefaultInteractiveSelect` for: project type, PHP, Node, package manager, DB driver, DB version, Redis version
- Use `pterm.DefaultInteractiveTextInput` for: domain, DB name
- DB version options switch dynamically based on selected DB driver

```go
package cmd

import (
	"fmt"
	"os"
	"path/filepath"
	"strings"

	"github.com/pterm/pterm"
	"github.com/spf13/cobra"
	"gopkg.in/yaml.v3"

	"github.com/XBS-Nathan/nova/internal/config"
)

func init() { rootCmd.AddCommand(initCmd) }

var initCmd = &cobra.Command{
	Use:   "init",
	Short: "Create a .nova/config.yaml for the current project",
	RunE: func(cmd *cobra.Command, args []string) error {
		cwd, err := os.Getwd()
		if err != nil {
			return fmt.Errorf("getting working directory: %w", err)
		}

		path := filepath.Join(cwd, config.ConfigFile)
		if _, err := os.Stat(path); err == nil {
			return fmt.Errorf("%s already exists — edit it directly or delete it first", config.ConfigFile)
		}

		// Create .nova/ directory
		novaDir := filepath.Join(cwd, ".nova")
		if err := os.MkdirAll(novaDir, 0755); err != nil {
			return fmt.Errorf("creating .nova directory: %w", err)
		}

		projectName := filepath.Base(cwd)

		// Auto-detect project type
		detectedType := "other"
		if _, err := os.Stat(filepath.Join(cwd, "artisan")); err == nil {
			detectedType = config.TypeLaravel
		}

		fmt.Println()
		pterm.DefaultSection.Printfln("nova init — %s", projectName)

		// --- General ---
		pterm.DefaultHeader.WithBackgroundStyle(pterm.NewStyle(pterm.BgDefault)).
			WithTextStyle(pterm.NewStyle(pterm.Bold)).Println("General")

		projectType, err := pterm.DefaultInteractiveSelect.
			WithOptions([]string{"laravel", "other"}).
			WithDefaultOption(detectedType).
			Show("Project type")
		if err != nil {
			return err
		}

		defaultDomain := projectName + ".test"
		domain, err := pterm.DefaultInteractiveTextInput.
			WithDefaultValue(defaultDomain).
			Show("Domain")
		if err != nil {
			return err
		}
		if !strings.HasSuffix(domain, ".test") {
			domain += ".test"
		}

		// --- Language ---
		pterm.DefaultHeader.WithBackgroundStyle(pterm.NewStyle(pterm.BgDefault)).
			WithTextStyle(pterm.NewStyle(pterm.Bold)).Println("Language")

		php, err := pterm.DefaultInteractiveSelect.
			WithOptions(config.PHPVersions).
			WithDefaultOption(config.DefaultPHP).
			Show("PHP version")
		if err != nil {
			return err
		}

		node, err := pterm.DefaultInteractiveSelect.
			WithOptions(config.NodeVersions).
			WithDefaultOption(config.DefaultNode).
			Show("Node version")
		if err != nil {
			return err
		}

		packageManager, err := pterm.DefaultInteractiveSelect.
			WithOptions([]string{"npm", "yarn", "pnpm"}).
			WithDefaultOption(config.DefaultPackageManager).
			Show("Package manager")
		if err != nil {
			return err
		}

		// --- Database ---
		pterm.DefaultHeader.WithBackgroundStyle(pterm.NewStyle(pterm.BgDefault)).
			WithTextStyle(pterm.NewStyle(pterm.Bold)).Println("Database")

		dbDriver, err := pterm.DefaultInteractiveSelect.
			WithOptions([]string{"mysql", "postgres"}).
			WithDefaultOption("mysql").
			Show("Driver")
		if err != nil {
			return err
		}

		dbVersionOptions := config.MySQLVersions
		defaultDBVer := config.DefaultMySQLVersion
		if dbDriver == "postgres" {
			dbVersionOptions = config.PostgresVersions
			defaultDBVer = config.DefaultPostgresVersion
		}

		dbVersion, err := pterm.DefaultInteractiveSelect.
			WithOptions(dbVersionOptions).
			WithDefaultOption(defaultDBVer).
			Show("Version")
		if err != nil {
			return err
		}

		defaultDBName := config.SanitizeName(projectName, true)
		dbName, err := pterm.DefaultInteractiveTextInput.
			WithDefaultValue(defaultDBName).
			Show("Database name")
		if err != nil {
			return err
		}

		// --- Services ---
		pterm.DefaultHeader.WithBackgroundStyle(pterm.NewStyle(pterm.BgDefault)).
			WithTextStyle(pterm.NewStyle(pterm.Bold)).Println("Services")

		redisVersion, err := pterm.DefaultInteractiveSelect.
			WithOptions(config.RedisVersions).
			WithDefaultOption(config.DefaultRedisVersion).
			Show("Redis version")
		if err != nil {
			return err
		}

		// Build config
		cfg := &initConfig{
			Domain:         domain,
			PHP:            php,
			Node:           node,
			PackageManager: packageManager,
			DBDriver:       dbDriver,
			DBVersion:      dbVersion,
			DB:             dbName,
			RedisVersion:   redisVersion,
		}
		if projectType != "other" {
			cfg.Type = projectType
		}

		data, err := yaml.Marshal(cfg)
		if err != nil {
			return fmt.Errorf("marshaling config: %w", err)
		}

		if err := os.WriteFile(path, data, 0644); err != nil {
			return fmt.Errorf("writing %s: %w", config.ConfigFile, err)
		}

		// Create .gitignore to exclude generated files but keep config
		gitignore := "*\n!config.yaml\n!.gitignore\n"
		if err := os.WriteFile(filepath.Join(novaDir, ".gitignore"), []byte(gitignore), 0644); err != nil {
			return fmt.Errorf("writing .nova/.gitignore: %w", err)
		}

		// Summary
		fmt.Println()
		pterm.DefaultSection.Println("Summary")
		fmt.Printf("  Type:     %s\n", projectType)
		fmt.Printf("  Domain:   %s\n", domain)
		fmt.Printf("  PHP:      %s\n", php)
		fmt.Printf("  Node:     %s (%s)\n", node, packageManager)
		fmt.Printf("  Database: %s %s (%s)\n", dbDriver, dbVersion, dbName)
		fmt.Printf("  Redis:    %s\n", redisVersion)
		fmt.Println()
		pterm.Success.Printfln("Created %s", config.ConfigFile)
		pterm.Info.Printfln("Run %s to get going.", pterm.LightCyan("nova start"))
		fmt.Println()

		return nil
	},
}

// initConfig controls which fields are written to .nova/config.yaml.
type initConfig struct {
	Type           string `yaml:"type,omitempty"`
	Domain         string `yaml:"domain"`
	PHP            string `yaml:"php"`
	Node           string `yaml:"node"`
	PackageManager string `yaml:"package_manager"`
	DBDriver       string `yaml:"db_driver"`
	DBVersion      string `yaml:"db_version"`
	DB             string `yaml:"db"`
	RedisVersion   string `yaml:"redis_version"`
}
```
